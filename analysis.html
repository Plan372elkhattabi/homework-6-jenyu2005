Question 1: Sample sizes by State

trees <- trees %>% #extract city name and abbreviate state name 
  mutate(
    city_name  = str_extract(City, "^[^,]+"), #extract everything before the comma (city name)
    state_abbr = str_extract(City, "[:upper:]{2}$") #extract 2 uppercase letters at the end (state abbreviation)
  )

#Count the records per state
state_counts <- trees %>%
  count(state_abbr, name = "n_records") %>%
  arrange(desc(n_records)) #sort from high to low (descending order)
